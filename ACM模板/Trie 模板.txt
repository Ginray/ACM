int flag;

typedef struct node  
{
	int data;
	struct node *childs[2];
	struct node ()
	{
	data=0;
	for(int i=0;i<2;i++)
		childs[i]=NULL;
	
	}

}node;

node *root,*cur,*newn;

void insert (char *x)
{
	int i,m;
	cur=root;

	 for(i=0;i<strlen(x);i++)  
    {  
        m=x[i]-'0';  
        if(cur->childs[m]!=NULL)  
        {  
            cur=cur->childs[m];  
            if(i==strlen(x)-1||cur->data==1)  
            {  
                flag=0;  
                break;  
            }  
        }  
        else  
        {  
            newn=new node;  
            cur->childs[m]=newn;  
            cur=newn;  
        }  
    }  
    cur->data=1;  


}
