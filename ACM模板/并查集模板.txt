int find (int x)
{
	int r =x;
	if(pre[r]!=r)

	pre[r]=find(pre[r]);

	return pre[r];

}


void join (int x,int y)

	}
//	fy = find(y);

		pre[fx]=pre[fy];
	return ;



}

void init(int n)
{
    int i;

    for(i=1;i<=2000+n;++i)pre[i] = i;
    flag = 0;
}
